  9def get_stiffness_tensor( pressure: List[List[List[float]]], deformation_magnitudes: List[float]  ):
 10    """
 11    Calculate the stiffness tensor based on the given pressure and deformation magnitudes.
 12
 13    Parameters:
 14        pressure (List[List[List[float]]]): A 3D list representing the pressure values for each deformation direction.
 15            The innermost list contains the pressure values for one deformation magnitude in the following order:
 16            [pxx, pyy, pzz, pyz, pxz, pxy].
 17            The middle list contains the pressure values for each deformation magnitude.
 18            The outermost list contains the pressure values for each deformation magnitude in each deformation direction (x,y,z,yz,xz,xy).
 19        deformation_magnitudes (List[float]): A list of deformation magnitudes.
 20
 21    Returns:
 22        numpy.ndarray: The stiffness tensor, a 6x6 numpy array.
 23            The entries of the stiffness tensor are calculated as follows:
 24            - C11: elongation in x direction and slope of pressure in x direction
 25            - C21: elongation in x direction and slope of pressure in y direction
 26            - C31: elongation in x direction and slope of pressure in z direction
 27            - C41: elongation in x direction and slope of pressure in yz direction
 28            - C51: elongation in x direction and slope of pressure in xz direction
 29            - C61: elongation in x direction and slope of pressure in xy direction
 30
 31    Note:
 32        - The stiffness tensor is calculated by performing a linear fit for each pressure along the deformation in each direction.
 33        - The stiffness tensor is symmetric, so the resulting tensor is rounded to ensure symmetry.
 34
 35    """
 36    ## Entries of stiffness tensor ##
 37
 38    # C11: elongation in x direction and slope of pressure in x direction
 39    # C21: elongation in x direction and slope of pressure in y direction
 40    # C31: elongation in x direction and slope of pressure in z direction 
 41    # C41: elongation in x direction and slope of pressure in yz direction 
 42    # C51: elongation in x direction and slope of pressure in xz direction 
 43    # C61: elongation in x direction and slope of pressure in xy direction 
 44
 45    stiffness_tensor = np.zeros((6,6))
 46
 47    for i in range(6):
 48        pxx,pyy,pzz = [p[0] for p in pressure[i]],[p[1] for p in pressure[i]],[p[2] for p in pressure[i]]
 49        pxy,pxz,pyz = [p[3] for p in pressure[i]],[p[4] for p in pressure[i]],[p[5] for p in pressure[i]]
 50
 51        ## (linear) Fit for each pressure along the deformation in xx,yy,zz,... direction ##
 52
 53        stiffness_tensor[0][i] = -np.polyfit( deformation_magnitudes, pxx, 1 )[0]
 54        stiffness_tensor[1][i] = -np.polyfit( deformation_magnitudes, pyy, 1 )[0]
 55        stiffness_tensor[2][i] = -np.polyfit( deformation_magnitudes, pzz, 1 )[0]
 56        stiffness_tensor[3][i] = -np.polyfit( deformation_magnitudes, pyz, 1 )[0]
 57        stiffness_tensor[4][i] = -np.polyfit( deformation_magnitudes, pxz, 1 )[0]
 58        stiffness_tensor[5][i] = -np.polyfit( deformation_magnitudes, pxy, 1 )[0] 
 59
 60    # Ensure symmetry
 61    stiffness_tensor = np.round( ( stiffness_tensor + stiffness_tensor.T ) / 2 , 3 )
 62
 63    return stiffness_tensor

 65def compute_VRH( stiffness_tensor: np.ndarray ):
 66    """
 67    Compute the Voigt-Reuss-Hill (VRH) averages of the elastic constants.
 68
 69    Parameters:
 70        stiffness_tensor (np.ndarray): The 6x6 stiffness tensor representing the elastic constants.
 71
 72    Returns:
 73        tuple: A tuple containing the VRH averages of the elastic constants in the following order:
 74            - K (float): Bulk modulus.
 75            - G (float): Shear modulus.
 76            - E (float): Young's modulus.
 77            - nu (float): Poisson's ratio.
 78
 79    Notes:
 80        - The stiffness tensor should be in Voigt notation.
 81        - The stiffness tensor should be symmetric.
 82        - The stiffness tensor should be positive definite.
 83
 84    References:
 85        - Voigt, W. (1928). Lehrbuch der Kristallphysik. Teubner.
 86        - Reuss, A. (1929). Berechnung der Fließgrenze von Mischkristallen auf Grund der Plastizitätsbedingung für Einkristalle. ZAMM - Journal of Applied Mathematics and Mechanics / Zeitschrift für Angewandte Mathematik und Mechanik, 9(1), 49–58.
 87        - Hill, R. (1952). The Elastic Behaviour of a Crystalline Aggregate. Proceedings of the Physical Society. Section A, 65(5), 349–354.
 88    """
 89    c_sq = ( stiffness_tensor[0][0] + stiffness_tensor[0][1] ) * stiffness_tensor[2][2] - 2 * stiffness_tensor[0][2]**2
 90    M    = stiffness_tensor[0][0] + stiffness_tensor[0][1] + 2 * stiffness_tensor[2][2] - 4 * stiffness_tensor[0][2]
 91
 92    K_V  = ( 1 / 9 * ( stiffness_tensor[0][0] + stiffness_tensor[1][1] + stiffness_tensor[2][2] ) + 
 93            2 / 9 * ( stiffness_tensor[0][1] + stiffness_tensor[0][2] + stiffness_tensor[1][2] ) )
 94
 95    K_R  = c_sq / M
 96
 97    G_V  = 1 / 30 * ( M + 12 * stiffness_tensor[3][3] + 12 * stiffness_tensor[5][5] ) 
 98
 99    G_R  = ( 5 / 2 * c_sq * stiffness_tensor[3][3] * stiffness_tensor[5][5] / 
100            ( c_sq * ( stiffness_tensor[3][3] + stiffness_tensor[5][5] ) + 3 * K_V * stiffness_tensor[3][3] * stiffness_tensor[5][5] ) )
101
102    K    = 0.5 * ( K_V + K_R )
103    G    = 0.5 * ( G_V + G_R )
104    E    = 9 * K * G / ( 3 * K + G )
105    nu   = (3 * K - 2 * G ) / ( 6 * K + 2 * G )
106
107    return K, G, E, nu

324    def analysis_lattice(self, supercell_dim: List[int], copies: int=-1, fraction: float=0.0, save_json: bool=True):
325        """
326        Performs lattice analysis on the equilibration simulations.
327
328        Parameters:
329            supercell_dim (List[int]): A list of three integers representing the dimensions of the supercell in the x, y, and z directions, respectively.
330            copies (int, optional): The number of equilibration copies to analyze. If set to -1, all equilibration copies will be analyzed. Defaults to -1.
331            fraction (float, optional): The fraction of data to use for analysis. Defaults to 0.0, which means all data will be used.
332            save_json (bool, optional): Flag indicating whether to save the analysis results as a JSON file. Defaults to True.
333
334        Returns:
335            dict: A dictionary containing the mean and standard deviation of the lattice parameters and density. The keys are:
336                - "a": [mean_a/fa, std_a/fa]
337                - "b": [mean_b/fb, std_b/fb]
338                - "c": [mean_c/fc, std_c/fc]
339                - "alpha": [mean_alpha, std_alpha]
340                - "beta": [mean_beta, std_beta]
341                - "gamma": [mean_gamma, std_gamma]
342                - "density": [mean_density, std_density]
343
344        Note:
345            - The lattice parameters are normalized by the dimensions of the supercell in the x, y, and z directions.
346            - The mean and standard deviation are calculated based on the equilibration copies specified or all equilibration copies if copies=-1.
347            - The fraction parameter can be used to analyze only a fraction of the data. If set to 0.0, all data will be used.
348            - If save_json is True, the analysis results will be saved as a JSON file in the equilibration folder.
349        """
350        if copies == -1:
351            copies = self.equilibration_copies
352
353        a_list = []
354        b_list = []
355        c_list = []
356        alpha_list = []
357        beta_list  = []
358        gamma_list = []
359        dens_list  = []
360
361        fa, fb, fc = supercell_dim
362
363        paths = [ f"{self.equilibration_folder}/0{copy}/{self.project_name}.lattice" for copy in range(copies+1) ] 
364
365        for path in paths:
366            _, a,b,c,alpha,beta,gamma,dens = read_lammps_output( path, keys=["v_a", "v_b", "v_c", "v_alpha" ,"v_beta", "v_gamma" ,"v_density"], fraction = fraction )
367            a_list.append(a)
368            b_list.append(b)
369            c_list.append(c)
370            alpha_list.append(alpha)
371            beta_list.append(beta)
372            gamma_list.append(gamma)
373            dens_list.append(dens)
374
375        results = { "a": [ np.mean(a_list) / fa, np.std( np.array(a_list) / fa ) ], "b": [ np.mean(b_list) / fb, np.std( np.array(b_list) / fb ) ],
376                    "c": [ np.mean(c_list) / fc, np.std( np.array(c_list) / fc ) ], "alpha": [ np.mean(alpha_list), np.std(alpha_list) ],
377                    "beta": [ np.mean(beta_list), np.std(beta_list) ], "gamma": [ np.mean(gamma_list), np.std(gamma_list) ], "density": [ np.mean(dens_list), np.std(dens_list) ] }
378        
379        if save_json:
380            work_json( file_path = f"{self.equilibration_folder}/lattice.json", data = results, to_do = "write" )
381
382        return results

384    def analysis_mechanical(self, deformation_magnitudes: List[float], copies: int=-1, fraction: float=0.0, save_json: bool=True, verbose: bool=False):
385        """
386        Performs mechanical analysis on the deformation simulations.
387
388        Parameters:
389            deformation_magnitudes (List[float]): A list of deformation magnitudes to be analyzed.
390            copies (int, optional): The number of deformation copies to analyze. If set to -1, all deformation copies will be analyzed. Defaults to -1.
391            fraction (float, optional): The fraction of data to use for analysis. Defaults to 0.0.
392            save_json (bool, optional): Flag indicating whether to save the analysis results as a JSON file. Defaults to True.
393            verbose (bool, optional): Flag indicating whether to print verbose output. Defaults to False.
394
395        Returns:
396            dict: A dictionary containing the analysis results. The keys are:
397                - "bulk_modulus": A list containing the mean and standard deviation of the bulk modulus.
398                - "shear_modulus": A list containing the mean and standard deviation of the shear modulus.
399                - "youngs_modulus": A list containing the mean and standard deviation of the Young's modulus.
400                - "poission_ratio": A list containing the mean and standard deviation of the Poisson's ratio.
401                - "stiffness_tensor": A list containing the mean and standard deviation of the stiffness tensor.
402
403        """
404        if copies == -1:
405            copies = self.deformation_copies
406
407        # Remove the undeformed state from all deformation directions
408        deformation_directions = [ "x", "y", "z", "yz", "xz", "xy", ]
409
410        # Result lists
411        p_main_e_main          = [[[],[],[]],[[],[],[]],[[],[],[]]]
412        stiffness_tensor_list  = []
413        K_list                 = []
414        G_list                 = []
415        E_list                 = []
416        nu_list                = []
417        results                = {}
418
419        for copy in range(copies+1):
420
421            #### Get data ####
422            
423            # Get the simulation paths
424            simulation_paths = [ [ f"{self.deformation_folder}/{deformation_direction}/{deformation_magnitude}/0{copy}/{self.project_name}.pressure" 
425                                   if deformation_magnitude != 0.0 else
426                                   f"{self.deformation_folder}/undeformed/0{copy}/{self.project_name}.pressure" 
427                                   for deformation_magnitude in deformation_magnitudes ] for deformation_direction in deformation_directions 
428                               ]
429
430            # Pre define pressure list
431            pressure = [ [ [] for _ in range( len( deformation_magnitudes ) ) ] for _ in range(6) ]
432
433            # Sample in each deformation direction (xx,yy,zz,yz,xz,xy) for each deformation (-0.05, ..., 0.00, ..., 0.05) 
434            # the resulting stress tensor (pxx,pyy,pzz,pxy,pxz,pyz)
435            for i,path in enumerate(simulation_paths):
436                # Convert pressure from atm in GPa ( GPa = 10^-9 Pa = 10^-9 * 101325 * atm )
437                for k,def_path in enumerate(path):
438                    press_array    = read_lammps_output( def_path, keys = [ "c_thermo_press[%d]"%j for j in range(1,7) ], fraction = fraction )
439                    # As the time average is also returned
440                    press_array    = press_array[1:]
441                    pressure[i][k] = press_array * 101325 / 1e9
442            
443            # Save the main stressses in the main derformation directions for visualisation purpose
444            for i in range(3):
445                p_main_e_main[i][0].append( [ p[0] for p in pressure[i] ] )
446                p_main_e_main[i][1].append( [ p[1] for p in pressure[i] ] )
447                p_main_e_main[i][2].append( [ p[2] for p in pressure[i] ] )
448            
449            #### Get slope of stress-strain curve at low deformation ####
450
451            stiffness_tensor = get_stiffness_tensor( pressure = pressure, deformation_magnitudes = deformation_magnitudes )
452
453            # Text ouput
454            txt  = f"\nCopy n°{copy}:\nResulting Stiffness Tensor\n"
455            txt += "\n"
456            for i in range(1,7): txt += "  ".join( [ "C%d%d"%(i,j) for j in range(1,7) ] ) + "\n"
457            txt += "\n"
458            for i in range(0,6): txt += "  ".join( [ str(np.round(st,2)) for st in stiffness_tensor[i,:] ] ) + "\n"
459            txt += "\n"
460
461            #### Computation of mechanical properties ####
462
463            #Voigt Reuss Hill
464            K, G, E, nu = compute_VRH( stiffness_tensor )
465
466            txt += "\nMechanical properties with Voigt Reuss Hill: \n"
467            txt += "Bulk modulus K = %.0f GPa\n"%K
468            txt += "Shear modulus G = %.0f GPa\n"%G
469            txt += "Youngs modulus E = %.0f GPa\n"%E
470            txt += "Poission ratio nu = %.3f\n"%nu
471
472            if verbose: print(txt)
473
474            stiffness_tensor_list.append( stiffness_tensor )
475            K_list.append( K )
476            G_list.append( G )
477            E_list.append( E )

 66def compute_VRH( stiffness_tensor: np.ndarray ):
 67    """
 68    Compute the Voigt-Reuss-Hill (VRH) averages of the elastic constants.
 69
 70    Parameters:
 71        stiffness_tensor (np.ndarray): The 6x6 stiffness tensor representing the elastic constants.
 72
 73    Returns:
 74        tuple: A tuple containing the VRH averages of the elastic constants in the following order:
 75            - K (float): Bulk modulus.
 76            - G (float): Shear modulus.
 77            - E (float): Young's modulus.
 78            - nu (float): Poisson's ratio.
 79
 80    Notes:
 81        - The stiffness tensor should be in Voigt notation.
 82        - The stiffness tensor should be symmetric.
 83        - The stiffness tensor should be positive definite.
 84
 85    References:
 86        - Voigt, W. (1928). Lehrbuch der Kristallphysik. Teubner.
 87        - Reuss, A. (1929). Berechnung der Fließgrenze von Mischkristallen auf Grund der Plastizitätsbedingung für Einkristalle. ZAMM - Journal of Applied Mathematics and Mechanics / Zeitschrift für Angewandte Mathematik und Mechanik, 9(1), 49–58.
 88        - Hill, R. (1952). The Elastic Behaviour of a Crystalline Aggregate. Proceedings of the Physical Society. Section A, 65(5), 349–354.
 89    """
 90    c_sq = ( stiffness_tensor[0][0] + stiffness_tensor[0][1] ) * stiffness_tensor[2][2] - 2 * stiffness_tensor[0][2]**2
 91    M    = stiffness_tensor[0][0] + stiffness_tensor[0][1] + 2 * stiffness_tensor[2][2] - 4 * stiffness_tensor[0][2]
 92
 93    K_V  = ( 1 / 9 * ( stiffness_tensor[0][0] + stiffness_tensor[1][1] + stiffness_tensor[2][2] ) + 
 94            2 / 9 * ( stiffness_tensor[0][1] + stiffness_tensor[0][2] + stiffness_tensor[1][2] ) )
 95
 96    K_R  = c_sq / M
 97
 98    G_V  = 1 / 30 * ( M + 12 * stiffness_tensor[3][3] + 12 * stiffness_tensor[5][5] ) 
 99
100    G_R  = ( 5 / 2 * c_sq * stiffness_tensor[3][3] * stiffness_tensor[5][5] / 
101            ( c_sq * ( stiffness_tensor[3][3] + stiffness_tensor[5][5] ) + 3 * K_V * stiffness_tensor[3][3] * stiffness_tensor[5][5] ) )
102
103    K    = 0.5 * ( K_V + K_R )
104    G    = 0.5 * ( G_V + G_R )
105    E    = 9 * K * G / ( 3 * K + G )
106    nu   = (3 * K - 2 * G ) / ( 6 * K + 2 * G )
107
108    return K, G, E, nu
